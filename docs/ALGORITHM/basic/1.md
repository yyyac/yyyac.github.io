## 快速排序

### [快速排序](https://www.acwing.com/problem/content/787/)

```C++
#include <iostream>
#include <algorithm>
#include <cstring>

using namespace std;

const int N = 100010;
int n;
int q[N];

void quick_sort(int l, int r)
{
    if (l >= r)
        return;
    int i = l - 1, j = r + 1, x = q[(l + r) / 2];
    while (i < j)
    {
        while (q[++i] < x);
        while (q[--j] > x);
        if (i < j)
            swap(q[i], q[j]);
    }
    quick_sort(l, j);
    quick_sort(j + 1, r);
}

int main()
{
    cin >> n;
    for (int i = 0; i < n; i++)
        cin >> q[i];
    quick_sort(0, n - 1);
    for (int i = 0; i < n; i++)
        cout << q[i] << " ";
    return 0;
}
```

### [第 k 个数](https://www.acwing.com/problem/content/788/)

```C++
#include <iostream>
#include <algorithm>
#include <cstring>

using namespace std;

const int N = 100010;
int n, k;
int q[N];

int find_K(int l, int r, int k)
{
    if (l == r)
        return q[l];
    int i = l - 1, j = r + 1, x = q[(l + r) / 2];
    while (i < j)
    {
        while (q[++i] < x);
        while (q[--j] > x);
        if (i < j)
            swap(q[i], q[j]);
    }
    int sl = j - l + 1;
    if (k <= sl)
        return find_K(l, j, k);
    return find_K(j + 1, r, k - sl);
}

int main()
{
    cin >> n >> k;
    for (int i = 0; i < n; i++)
        cin >> q[i];
    cout << find_K(0, n - 1, k) << endl;
    return 0;
}
```

## 归并排序

### [归并排序](https://www.acwing.com/problem/content/789/)

```C++
#include <iostream>
#include <algorithm>
#include <cstring>

using namespace std;

const int N = 100010;
int n;
int a[N], b[N];

void merge_sort(int l, int r)
{
    if (l >= r)
        return;
    int mid = (l + r) / 2;
    merge_sort(l, mid), merge_sort(mid + 1, r);
    int k = 0, i = l, j = mid + 1;
    while (i <= mid && j <= r)
    {
        if (a[i] <= a[j])
            b[k++] = a[i++];
        else
            b[k++] = a[j++];
    }
    while (i <= mid)
        b[k++] = a[i++];
    while (j <= r)
        b[k++] = a[j++];

    for (int i = l, j = 0; i <= r; i++, j++)
        a[i] = b[j];
}

int main()
{
    cin >> n;
    for (int i = 0; i < n; i++)
        cin >> a[i];
    merge_sort(0, n - 1);
    for (int i = 0; i < n; i++)
        cout << a[i] << " ";
    return 0;
}
```

## 二分

### [数的范围](https://www.acwing.com/problem/content/791/)

```C++
#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;
const int N = 100010;
int n, m;
int a[N];

int find1(int n, int x)
{
    int l = -1, r = n;
    while(l + 1 != r)
    {
        int mid = (l + r) / 2;
        if(a[mid] < x) l = mid;
        else r = mid;
    }
    if(a[r] == x) return r;
    return -1;
}

int find2(int n, int x)
{
    int l = -1, r = n;
    while(l + 1 != r)
    {
        int mid = (l + r) / 2;
        if(a[mid] <= x) l = mid;
        else r = mid;
    }
    if(a[l] == x) return l;
    return -1;
}

int main()
{
    cin >> n >> m;
    for (int i = 0; i < n; i ++ )
        cin >> a[i];
    while (m -- )
    {
        int x;
        cin >> x;
        int res1 = find1(n, x);
        int res2 = find2(n, x);
        cout << res1 << " " << res2 << endl;
    }
    return 0;
    
}
```
